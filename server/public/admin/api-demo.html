কোড উদাহরণ: শিক্ষার্থী তালিকা

পুরাতন পদ্ধতি (StorageManager):
// পুরাতন পদ্ধতি - লোকাল স্টোরেজ থেকে ডেটা নেওয়া // সকল শিক্ষার্থী পড়ুন const students = StorageManager.getStudents(); console.log(students); // আইডি দিয়ে শিক্ষার্থী খুঁজুন const student = StorageManager.getStudentById(1); // নতুন শিক্ষার্থী যোগ করুন const newStudent = StorageManager.addStudent({ firstName: 'আব্দুল', lastName: 'রহমান', classId: 1 }); // শিক্ষার্থী আপডেট করুন StorageManager.updateStudent(1, { status: 'active' }); // শিক্ষার্থী মুছুন StorageManager.deleteStudent(1);
নতুন পদ্ধতি (API) - অ্যাসিঙ্ক/অয়েট সাপোর্ট:
// নতুন পদ্ধতি - REST API থেকে ডেটা নেওয়া // সকল শিক্ষার্থী পড়ুন (অ্যাসিঙ্ক) async function loadStudents() { try { const students = await API.students.getAll(); console.log(students); return students; } catch (error) { console.error('এরর:', error); } } // আইডি দিয়ে শিক্ষার্থী খুঁজুন async function getStudent(id) { const student = await API.students.getById(id); return student; } // নতুন শিক্ষার্থী যোগ করুন async function createStudent(data) { const newStudent = await API.students.create(data); console.log('শিক্ষার্থী যোগ হয়েছে:', newStudent); return newStudent; } // শিক্ষার্থী আপডেট করুন async function updateStudent(id, data) { const updated = await API.students.update(id, data); showToast('আপডেট সফল হয়েছে', 'success'); return updated; } // শিক্ষার্থী মুছুন async function deleteStudent(id) { await API.students.delete(id); showToast('শিক্ষার্থী মুছে ফেলা হয়েছে', 'success'); } // সার্চ করুন async function searchStudents(query) { const results = await API.students.search(query); return results; } // ক্লাস অনুযায়ী শিক্ষার্থী async function getStudentsByClass(classId, section) { const students = await API.students.getByClass(classId, section); return students; }

ব্যবহার নির্দেশনা

  • ধাপ ১: প্রথমে Node.js সার্ভার চালু করুন:
    npm run dev অথবা node server/index.js
  • ধাপ ২: HTML ফাইলে স্ক্রিপ্ট যোগ করুন:
    <script src="js/api.js"></script> <script src="js/data-manager.js"></script>
  • ধাপ ৩: পেজ লোডে ডেটা ম্যানেজার ইনিশিয়ালাইজ করুন:
    document.addEventListener('DOMContentLoaded', async () => { await DataManager.init({ mode: 'hybrid' }); loadStudents(); });
  • ধাপ ৪: স্টোরেজ ম্যানেজার কলগুলো API দিয়ে রিপ্লেস করুন:
    // পুরাতন: const students = StorageManager.getStudents(); // নতুন: const students = await API.students.getAll();
